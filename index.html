            <input type="text" id="taskInput" placeholder="O que você vai fazer hoje?">

            <button onclick="addTask()">+ Tarefa</button>

        </div>

        <ul id="taskList"></ul>

        <div id="emptyMessage" class="empty-message">Tudo limpo por aqui, Evelyn! ✨</div>

    </div>

    <!-- Bloco 2: Recompensas -->

    <div class="container">

        <h2>Loja de Recompensas</h2>

       

        <div class="input-group reward-input-group">

            <input type="text" id="rewardInput" placeholder="Ex: Assistir série, doce...">

            <input type="number" id="rewardCostInput" placeholder="Custo" min="10" step="10" value="50">

            <button onclick="addReward()">+ Prêmio</button>

        </div>

        <ul id="rewardList"></ul>

        <div id="emptyRewardMessage" class="empty-message">Nenhum prêmio cadastrado ainda! 🎁</div>

    <script>

        let score = 0;

        const taskInput = document.getElementById('taskInput');

        const taskList = document.getElementById('taskList');

        const emptyMessage = document.getElementById('emptyMessage');

        const scoreDisplay = document.getElementById('scoreDisplay');

        const rewardInput = document.getElementById('rewardInput');

        const rewardCostInput = document.getElementById('rewardCostInput');

        const rewardList = document.getElementById('rewardList');

        const emptyRewardMessage = document.getElementById('emptyRewardMessage');

        // Inicialização

        document.addEventListener('DOMContentLoaded', () => {

            loadData();

            checkEmptyTasks();

            checkEmptyRewards();

        });

        // Eventos de teclado (Enter)

        taskInput.addEventListener('keypress', e => { if (e.key === 'Enter') addTask(); });

        rewardInput.addEventListener('keypress', e => { if (e.key === 'Enter') addReward(); });

        // --- SISTEMA DE TAREFAS ---

        function addTask() {

            const text = taskInput.value.trim();

            if (!text) return alert('Digite a tarefa!');

            createTaskElement(text, false);

            taskInput.value = '';

            saveData();

            checkEmptyTasks();

        }

        function createTaskElement(text, isCompleted) {

            const li = document.createElement('li');

            if (isCompleted) li.classList.add('completed');

            const span = document.createElement('span');

            span.className = 'task-text';

            span.innerText = text;

           

            // Logica de clicar para concluir (+10 ou -10 pts)

            span.onclick = function() {

                if (li.classList.contains('completed')) {

                    li.classList.remove('completed');

                    updateScore(-10);

                } else {

                    li.classList.add('completed');

                    updateScore(10);

                }

                saveData();

            };

            const deleteBtn = document.createElement('button');

            deleteBtn.className = 'delete-btn';

            deleteBtn.innerText = 'X';

            deleteBtn.onclick = function() {

                // Se deletar uma tarefa já concluída, reduz os pontos dela

                if (li.classList.contains('completed')) {

                    updateScore(-10);

                }

                li.remove();

                saveData();

                checkEmptyTasks();

            };

            li.appendChild(span);

            li.appendChild(deleteBtn);

            taskList.appendChild(li);

        }

        // --- SISTEMA DE RECOMPENSAS ---

        function addReward() {

            const text = rewardInput.value.trim();

            const cost = parseInt(rewardCostInput.value);

            if (!text || isNaN(cost) || cost <= 0) return alert('Insira um prêmio e um custo válido!');

            createRewardElement(text, cost);

            rewardInput.value = '';

            rewardCostInput.value = '50';

            saveData();

            checkEmptyRewards();

        }

        function createRewardElement(text, cost) {

            const li = document.createElement('li');

            const spanText = document.createElement('span');

            spanText.className = 'reward-text';

            spanText.innerText = text;

            const badge = document.createElement('span');

            badge.className = 'pts-badge';

            badge.innerText = `${cost} XP`;

            const actionsDiv = document.createElement('div');

            actionsDiv.className = 'actions';

            // Botão Resgatar

            const claimBtn = document.createElement('button');

            claimBtn.className = 'claim-btn';

            claimBtn.innerText = 'Resgatar';

            claimBtn.onclick = function() {

                if (score >= cost) {

                    updateScore(-cost);

                    alert(`Parabéns Evelyn! Você resgatou: "${text}" 🎉 Aproveite!`);

                    saveData();

                } else {

                    alert(`Pontos insuficientes! Você precisa de mais ${cost - score} XP para este prêmio.`);

                }

            };

            // Botão Excluir Prêmio

            const deleteBtn = document.createElement('button');

            deleteBtn.className = 'delete-btn';

            deleteBtn.innerText = 'X';

            deleteBtn.onclick = function() {

                li.remove();

                saveData();

                checkEmptyRewards();

            };

            actionsDiv.appendChild(claimBtn);

            actionsDiv.appendChild(deleteBtn);

            li.appendChild(badge);

            li.appendChild(spanText);

            li.appendChild(actionsDiv);

            rewardList.appendChild(li);

        }

        // --- CONTROLE DE PONTOS E DADOS ---

        function updateScore(value) {

            score += value;

            if (score < 0) score = 0; // Evita pontos negativos

            scoreDisplay.innerText = `${score} XP`;

        }

        function saveData() {

            // Salva Pontos

            localStorage.setItem('evScore', score);

            // Salva Tarefas

            const tasks = [];

            document.querySelectorAll('#taskList li').forEach(li => {

                tasks.push({

                    text: li.querySelector('.task-text').innerText,

                    completed: li.classList.contains('completed')

                });

            });

            localStorage.setItem('evTasksGame', JSON.stringify(tasks));

            // Salva Recompensas

            const rewards = [];

            document.querySelectorAll('#rewardList li').forEach(li => {

                const costBadge = li.querySelector('.pts-badge').innerText;

                rewards.push({

                    text: li.querySelector('.reward-text').innerText,

                    cost: parseInt(costBadge)

                });

            });

            localStorage.setItem('evRewardsGame', JSON.stringify(rewards));

        }

        function loadData() {

            // Carrega Pontos

            score = parseInt(localStorage.getItem('evScore')) || 0;

            scoreDisplay.innerText = `${score} XP`;

            // Carrega Tarefas

            const savedTasks = JSON.parse(localStorage.getItem('evTasksGame')) || [];

            savedTasks.forEach(task => createTaskElement(task.text, task.completed));

            // Carrega Recompensas

            const savedRewards = JSON.parse(localStorage.getItem('evRewardsGame')) || [];

            savedRewards.forEach(reward => createRewardElement(reward.text, reward.cost));

        }

        function checkEmptyTasks() {

            emptyMessage.style.display = taskList.children.length === 0 ? 'block' : 'none';

        }

        function checkEmptyRewards() {

            emptyRewardMessage.style.display = rewardList.children.length === 0 ? 'block' : 'none';

        }

    </script>
